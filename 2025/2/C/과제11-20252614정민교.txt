#include <stdio.h>

int tripleValue(int a)
{
    return a * 3;
}

int* triplePointer1(int *a)
{
    return *a * 3;
}

int main()
{
    int a = 3;

    printf("%d\n", tripleValue(a));
    printf("%d\n", triplePointer1(&a));
}

#include <stdio.h>

void swap(int *a, int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main()
{
    int a, b;
    scanf("%d %d", &a, &b);
    swap(&a,&b);
    printf("%d %d", a, b);
}

#include <stdio.h>

void cubic(int *a, int *b)
{
    *a *= *a * *a;
    *b *= *b * *b;
}

int main()
{
    int a, b;
    scanf("%d %d", &a, &b);
    cubic(&a, &b);
    printf("%d %d", a, b);
}

#include <stdio.h>

void fuc(int *a, int *b, int *c)
{
    if(*a < 0)
    {
        *a = 0;
    }
    if(*b < 0)
    {
        *b = 0;
    }
    if(*c < 0)
    {
        *c = 0;
    }
}

int main()
{
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);

    fuc(&a,&b,&c);

    printf("%d %d %d", a, b, c);
}
