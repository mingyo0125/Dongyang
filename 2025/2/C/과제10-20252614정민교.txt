#include <stdio.h>

int sum_array(int arr[], int n) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += arr[i];
    }
    return sum;
}

int main(void) {
    int n;
    printf("배열 크기를 입력하세요: ");
    scanf("%d", &n);

    int arr[100];   // 최대 100개라고 가정
    if (n > 100) n = 100;

    printf("%d개의 정수를 입력하세요:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int sum = sum_array(arr, n);
    double avg = sum / (double)n;

    printf("합 = %d, 평균 = %.2f\n", sum, avg);

    return 0;
}

#include <stdio.h>

int max_array(int arr[], int n) {
    int max = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

int main(void) {
    int n;
    printf("배열 크기를 입력하세요: ");
    scanf("%d", &n);

    int arr[100];
    if (n > 100) n = 100;

    printf("%d개의 정수를 입력하세요:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int max = max_array(arr, n);
    printf("최댓값 = %d\n", max);

    return 0;
}

#include <stdio.h>

void reverse_array(int arr[], int n) {
    int temp;
    for (int i = 0; i < n / 2; i++) {
        temp = arr[i];
        arr[i] = arr[n - 1 - i];
        arr[n - 1 - i] = temp;
    }
}

int main(void) {
    int n;
    printf("배열 크기를 입력하세요: ");
    scanf("%d", &n);

    int arr[100];
    if (n > 100) n = 100;

    printf("%d개의 정수를 입력하세요:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    reverse_array(arr, n);

    printf("역순 배열: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    return 0;
}

#include <stdio.h>

int is_right_triangle(int a, int b, int c) {
    return (a*a + b*b == c*c);
}

int main(void) {
    int count = 0;

    for (int a = 1; a <= 50 && count < 10; a++) {
        for (int b = a; b <= 50 && count < 10; b++) {  
            for (int c = b; c <= 50 && count < 10; c++) {
                if (is_right_triangle(a, b, c)) {
                    printf("%d세트: (%d, %d, %d)\n", count + 1, a, b, c);
                    count++;
                }
            }
        }
    }

    if (count < 10) {
        printf("지정 범위 내에서 10세트를 찾지 못했습니다.\n");
    }

    return 0;
}

double inchTocenti(double inch){
	double centi = inch * 2.54;
