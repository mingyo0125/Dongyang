#define SIZE 30
int main()
{
    int arr[SIZE];
    int sum = 0;
    for(int i = 0; i < SIZE; i++)
    {
        arr[i] = 3 * (i + 1);

        sum += arr[i];

    printf("%d", sum);

#include <stdio.h>
#include <stdlib.h>

#define ROWSIZE 3
#define COLSIZE 4
int main()
{
    int arr[ROWSIZE][COLSIZE];
    int sum = 0;

    for(int i = 0; i < ROWSIZE; i++)
    {
        for(int j = 0; j < COLSIZE; j++)
        {
            arr[i][j] = rand();
        }
    }

    for(int i = 0; i < ROWSIZE; i++)
    {
        for(int j = 0; j < COLSIZE; j++)
        {
            sum += arr[i][j];
        }
    }

    printf("%lf", sum / (ROWSIZE * COLSIZE));
}

{

    {
        scanf("%d", &arr[i]);
    }

    int max = arr[0];

    for(int i = 0; i < SIZE; i++)
    {
        if(max < arr[i])
        {
            max = arr[i];
