#include <stdio.h>

int volume(int w, int l, int h)
{
    return w * l * h;
}

int surface(int w, int l, int h)
{
    return (2 * w * l) + (2 * l * h) + (2 * w * h);
}

int main()
{
    int w, l, h;
    scanf("%d %d %d", &w, &l, &h);

    printf("%d", volume(w, l, h));
    printf("%d", surface(w, l, h));
}

{

#include <stdio.h>

int Max(int a, int b, int c)
{
    int max = a;
    if(b > max)
    {
        max = b;
    }
    if(c > max)
    {
        max = c;
    }

    return max;
}

int Min(int a, int b, int c)
{
    int min = a;
    if(b < min)
    {
        min = b;
    }
    if(c < min)
    {
        min = c;
    }

    return min;
}

int main()
{
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);

    printf("%d", Max(a,b,c) * Min(a,b,c));
}

#include <stdio.h>

int Sum(int a, int b)
{
    int start, end;
    if(a > b)
    {
        start = b;
        end = a;
    }
    else
    {
        start = a;
        end = b;
    }

    int sum = 0;

    for(int i = start; i <= end; i++)
    {
        sum += i;
    }

    return sum;
}

int main()
{
    int a, b;

    scanf("%d %d", &a, &b);

    printf("%d", Sum(a,b));
}

#include <stdio.h>

int is_leap(int year)
{
    if (year % 400 == 0)
    {
        return 1;
    }
    if (year % 100 == 0)
    {
        return 0;
    }
    if (year % 4 == 0)
    {
        return 1;
    }
    return 0;
}

int main() {
    int birth_year;
    int total_days = 0;

    scanf("%d", &birth_year);

    for (int i = birth_year; i <= 2030; i++)
    {
        total_days += 365 + is_leap(i);
    }

    printf("%d", total_days);
}

#include <stdio.h>

int isPrime(int a)
{
    for(int i = 2; i < a; i++)
    {
        if(a % i == 0)
        {
            return 0;
        }
    }
    return 1;
}

int main()
{
    int a;
    scanf("%d", &a);
    int primeNum = 0;
    for(int i = a - 1; i > 1; i--)
    {
        if(isPrime(i) == 1)
        {
            primeNum = i;
            break;
        }
    }

    printf("%d", primeNum);
}

int isequalarray(int a[], int b[], int n)
{
	for (int i = 0; i < n; i++)
    {
		if (a[i] != b[i])
        {
            return 0;
        }
	}
	return 1;
}
